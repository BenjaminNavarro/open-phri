function(add_application app)
  PID_Component(
    EXAMPLE_APPLICATION
    NAME ${app}
    DIRECTORY ${app}
    RUNTIME_RESOURCES ${app}
    DEPEND 
      open-phri/open-phri
      open-phri/open-phri-vrep-driver
      pid-os-utilities/pid-signal-manager
  )
endfunction()
